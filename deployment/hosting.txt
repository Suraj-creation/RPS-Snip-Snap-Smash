


1. Create Self Signed Certificates 

sudo mkdir -p /etc/nginx/ssl
sudo openssl req -x509 -nodes -days 365 \
  -newkey rsa:2048 \
  -keyout /etc/nginx/ssl/selfsigned.key \
  -out /etc/nginx/ssl/selfsigned.crt


2. Nginx Configuration
/etc/nginx/conf.d/rps.conf


--content-begin--
server {
    listen 80 default_server;
    server_name _;

    location / {
        proxy_pass http://127.0.0.1:8000;
    }
}

server {
    listen 443 ssl default_server;
    server_name _;

    ssl_certificate /etc/nginx/ssl/selfsigned.crt;
    ssl_certificate_key /etc/nginx/ssl/selfsigned.key;

    location / {
        proxy_pass http://127.0.0.1:8000;

        proxy_http_version 1.1;
        proxy_set_header Host $host;
        proxy_set_header Upgrade $http_upgrade;
        proxy_set_header Connection "upgrade";
    }
}
--content-end--



sudo systemctl restart nginx

3. Linux Service Unit.

/etc/systemd/system/rps.service

--content-begin--
[Unit]
Description=RPS
After=network.target

[Service]
User=venkatababjisama
WorkingDirectory=/home/venkatababjisama/rps/vu-rsp-dc-main/deployment/src/server

# Activate virtual environment
ExecStart=/home/venkatababjisama/pyenv/bin/uvicorn main:app \
    --host 0.0.0.0 \
    --port 8000

Restart=always
RestartSec=3

Environment="PATH=/home/venkatababjisama/pyenv/bin/"
Environment="PYTHONUNBUFFERED=1"

[Install]
WantedBy=multi-user.target
Set the appropriate user and path. 

sudo systemctl daemon-reload
sudo systemctl start rps
sudo systemctl enable rps
sudo systemctl status rps
--content-end--
